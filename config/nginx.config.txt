upstream puma {
  server unix:///home/deploy/football/shared/tmp/sockets/puma.sock;
}

upstream puma2 {
  server unix:///home/deploy/price/shared/tmp/sockets/puma.sock;
}

server {
  listen 443 ssl;
  server_name krsz.ru;

  root       /home/deploy/football/current/public;
  access_log /home/deploy/football/current/log/nginx.access.log;
  error_log  /home/deploy/football/current/log/nginx.error.log info;

  location ^~ /assets/ {
    gzip_static on;
    expires max;
    add_header Cache-Control public;
  }

  try_files $uri/index.html $uri @puma;
  location @puma {
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_set_header Host $http_host;
    proxy_redirect off;
    proxy_pass http://puma;
  }

  error_page 500 502 503 504 /500.html;
  client_max_body_size 10M;
  keepalive_timeout 10;
  ssl_certificate /etc/letsencrypt/live/krsz.ru/fullchain.pem;
  ssl_certificate_key /etc/letsencrypt/live/krsz.ru/privkey.pem;
}
